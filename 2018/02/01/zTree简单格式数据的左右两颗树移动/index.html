zTree简单格式数据的左右两颗树移动

有一个需求是，使用ztree来实现左右两棵树的节点移动和移除。

大概就是如图所示：

选择左侧树，点击增加按钮节点移动到右侧。

选择右侧树，点击移出按钮节点移动到左侧。

这里包含了一个关键点就是父节点的删除与否，如果该父节点下的子节点都被移出，就要删除该父节点。否则保留。

既然这样，还不如直接上代码。然后有需要的各位可以直接运行代码。跑一下看下效果嘛。

        function addRole(){
            var Tree1 = $.fn.zTree.getZTreeObj("treeDemo");
            var Tree2 = $.fn.zTree.getZTreeObj("treeDemo2");
            moveTreeNode(Tree1,Tree2)
        }

        function delRole() {
            var Tree1 = $.fn.zTree.getZTreeObj("treeDemo");
            var Tree2 = $.fn.zTree.getZTreeObj("treeDemo2");
            moveTreeNode(Tree2,Tree1)
        }

        function moveTreeNode(Tree1,Tree2) {
            var nodes = Tree1.getCheckedNodes(); //获取到当前选中的元素，

            $.each(nodes,function (i,node) {
                var strs={};
                strs.id =node.id;
                strs.bomCn=node.bomCn;
                strs.parentId= node.parentId;
                // 增加新的
                zTreeDataAddNode(strs,Tree2);
                // 删除旧的
                zTreeDataDelete(node,Tree1);
            });
        }

        function zTreeDataDelete(node,Tree1) {
            var halfCheck = node.getCheckStatus();
            // 只删除全选状态，半选的不删
            if (!halfCheck.half){
                Tree1.removeNode(node);
            }
        }

        
        function zTreeDataAddNode(strs,Tree2) {

            var nodes =Tree2.transformToArray(Tree2.getNodes());
            if(nodes.length > 0){
                //这个循环判断是否已经存在,true表示不存在可以添加,false存在不能添加
                var is_add=true;
                for(var j=0;j<nodes.length;j++){
                    if(strs.id==nodes[j].id){
                        is_add=false;
                        break;
                    }
                }
                var parentId = strs.parentId;
                // 寻找父亲节点，如果有父节点返回true 说明需要在父节点上增加子节点
                if(parentId == null){
                    var flag =false;
                } else {
                    var k=0;
                    var flag =false;
                    for(k;k<nodes.length;k++){
                        if(parentId == nodes[k].id){
                            flag = true;
                            break;
                        }
                    }
                }
                if(flag && is_add){
                    // 添加子节点
                    var treeNode = nodes[k];
                    Tree2.addNodes(treeNode,strs);
                }else if(is_add){
                    // 添加父节点
                    Tree2.addNodes(null,strs);
                }
            } else {
                //树没任何数据时,第一个被加进来的元素
                Tree2.addNodes(null,strs);
            }
        }
